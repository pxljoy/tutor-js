      //Instantiate a Tutor object
      var tutorial = new Tutor;

      //Add steps
      tutorial
      .addStep('#step1', {
        on:'click',
        class:'example--current',
        start:function(){
          $('#example-start').text('Running...');
          $('#tutorial-title').text('Headings');
          $('#tutorial-content').text('This is a heading element. Click it to continue.')
          $('#tutorial').fadeIn(150);
        },
      })
      .addStep('#step2', {
        on:'dblclick',
        class:'example--current',
        start:function(){
          $('#tutorial-title').text('Paragraph');
          $('#tutorial-content').text('This is a paragraph element. Double click it to continue.')
          $('#tutorial').fadeIn(150);
        },
      });

      $('#example-start').click(function(){
        tutorial.start(null, function(){
          $('#example-start').text('Done! Restart?');
          $('#tutorial').fadeOut(150);
        });
      });
